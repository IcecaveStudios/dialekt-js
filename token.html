      @typeDescription: (type) ->
    switch type
      when @LOGICAL_AND
        return 'AND operator'
      when @LOGICAL_OR
        return 'OR operator'
      when @LOGICAL_NOT
        return 'NOT operator'
      when @STRING
        return 'tag'
      when @OPEN_BRACKET
        return 'open bracket'
      when @CLOSE_BRACKET
        return 'close bracket'

    throw new Error("Dialekt Token: Unknown type.")


module.exports = Token
